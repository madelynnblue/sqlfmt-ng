# sqlfmtng Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a new Rust SQL formatter at `~/code/sqlfmtng` supporting multiple dialects (Materialize and Postgres in Phase 1) via a common typed IR, with CLI and browser WASM entry points.

**Architecture:** Each dialect crate parses SQL using its native parser and returns a typed `Node` enum (defined in `sqlfmt-ir`). The renderer (`sqlfmt-render`) walks the `Node` tree, applies case transformation, and converts to `RcDoc` from the `pretty` crate (Wadler-Lindig algorithm). CockroachDB and GoogleSQL are stubbed as `unimplemented!()` for Phase 2.

**Tech Stack:** Rust workspace, `pretty` v0.12 (Wadler layout), `serde`/`serde_json` v1, `mz-sql-parser` (Materialize AST), `pg_query` v5 (Postgres AST), `clap` v4 (CLI), `wasm-bindgen` (browser WASM), `wasm-pack` (browser build).

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Clause {
    pub keyword: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub body: Option<Box<Node>>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "snake_case")]
pub enum Node {
    // Sugar nodes
    Clauses {
        items: Vec<Clause>,
    },
    List {
        items: Vec<Node>,
        #[serde(skip_serializing_if = "Option::is_none")]
        separator: Option<String>,
    },
    Wrap {
        #[serde(skip_serializing_if = "Option::is_none")]
        keyword: Option<String>,
        open: String,
        content: Box<Node>,
        close: String,
    },
    Infix {
        op: String,
        items: Vec<Node>,
    },
    Keyword {
        value: String,
    },
    Identifier {
        value: String,
        #[serde(skip_serializing_if = "Option::is_none")]
        quote: Option<String>,
    },
    Literal {
        value: String,
    },
    // Primitives
    Text {
        value: String,
    },
    Concat {
        items: Vec<Node>,
    },
    Group {
        content: Box<Node>,
    },
    Nest {
        content: Box<Node>,
    },
    Line,
    Hardline,
}

// ~/code/sqlfmtng/crates/sqlfmt-render/src/lib.rs
use pretty::RcDoc;
use sqlfmt_ir::{Clause, Node};

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum CaseMode {
    Upper,
    Lower,
    Title,
}

#[derive(Debug, Clone)]
pub struct RenderOpts {
    pub line_width: usize,
    pub use_tabs: bool,
    pub tab_width: usize,
    pub case: CaseMode,
}

impl Default for RenderOpts {
    fn default() -> Self {
        Self {
            line_width: 60,
            use_tabs: false,
            tab_width: 4,
            case: CaseMode::Upper,
        }
    }
}

fn apply_case(s: &str, mode: CaseMode) -> String {
    match mode {
        CaseMode::Upper => s.to_uppercase(),
        CaseMode::Lower => s.to_lowercase(),
        CaseMode::Title => {
            let lower = s.to_lowercase();
            let mut chars = lower.chars();
            match chars.next() {
                None => String::new(),
                Some(c) => c.to_uppercase().collect::<String>() + chars.as_str(),
            }
        }
    }
}

pub fn render(node: &Node, opts: &RenderOpts) -> String {
    let doc = to_rdoc(node, opts);
    let mut w = Vec::new();
    doc.render(opts.line_width, &mut w).unwrap();
    String::from_utf8(w).unwrap()
}

fn to_rdoc<'a>(node: &Node, opts: &RenderOpts) -> RcDoc<'a> {
    match node {
        Node::Clauses { items } => {
            let docs: Vec<RcDoc> = items.iter().map(|c| clause_to_rdoc(c, opts)).collect();
            RcDoc::intersperse(docs, RcDoc::hardline())
        }
        Node::List { items, separator } => {
            let sep_str = separator.as_deref().unwrap_or(",");
            let sep = RcDoc::text(sep_str.to_owned()).append(RcDoc::line());
            let docs: Vec<RcDoc> = items.iter().map(|n| to_rdoc(n, opts)).collect();
            RcDoc::group(RcDoc::intersperse(docs, sep))
        }
        Node::Wrap { keyword, open, content, close } => {
            let prefix: RcDoc = if let Some(kw) = keyword {
                RcDoc::text(apply_case(kw, opts.case))
                    .append(RcDoc::text(open.clone()))
            } else {
                RcDoc::text(open.clone())
            };
            prefix
                .append(to_rdoc(content, opts))
                .append(RcDoc::text(close.clone()))
        }
        Node::Infix { op, items } => {
            let sep = RcDoc::line()
                .append(RcDoc::text(op.clone()))
                .append(RcDoc::space());
            let docs: Vec<RcDoc> = items.iter().map(|n| to_rdoc(n, opts)).collect();
            RcDoc::group(RcDoc::intersperse(docs, sep))
        }
        Node::Keyword { value } => RcDoc::text(apply_case(value, opts.case)),
        Node::Identifier { value, quote } => match quote {
            Some(q) => RcDoc::text(format!("{q}{value}{q}")),
            None => RcDoc::text(value.clone()),
        },
        Node::Literal { value } => RcDoc::text(value.clone()),
        Node::Text { value } => RcDoc::text(value.clone()),
        Node::Concat { items } => {
            let docs: Vec<RcDoc> = items.iter().map(|n| to_rdoc(n, opts)).collect();
            RcDoc::concat(docs.into_iter())
        }
        Node::Group { content } => RcDoc::group(to_rdoc(content, opts)),
        Node::Nest { content } => {
            RcDoc::nest(opts.tab_width as isize, to_rdoc(content, opts))
        }
        Node::Line => RcDoc::line(),
        Node::Hardline => RcDoc::hardline(),
    }
}

fn clause_to_rdoc<'a>(clause: &Clause, opts: &RenderOpts) -> RcDoc<'a> {
    let kw = RcDoc::text(apply_case(&clause.keyword, opts.case));
    match &clause.body {
        None => kw,
        Some(body) => {
            let body_doc = to_rdoc(body, opts);
            RcDoc::group(
                kw.append(RcDoc::line())
                    .append(RcDoc::nest(opts.tab_width as isize, body_doc)),
            )
        }
    }
}
```

// ~/code/sqlfmtng/crates/sqlfmt-core/src/lib.rs
use sqlfmt_ir::Node;
use sqlfmt_render::{render, RenderOpts};
use thiserror::Error;

#[derive(Debug, Error)]
pub enum SqlfmtError {
    #[error("parse error: {0}")]
    Parse(String),
}

pub trait Dialect {
    fn parse(&self, sql: &str) -> Result<Node, SqlfmtError>;
}

pub fn format_sql(
    dialect: &dyn Dialect,
    sql: &str,
    opts: &RenderOpts,
) -> Result<String, SqlfmtError> {
    let node = dialect.parse(sql)?;
    Ok(render(&node, opts))
}
```

use mz_sql_parser::ast::*;
use sqlfmt_ir::{Clause, Node};

pub fn statement_to_node(stmt: &Statement<Raw>) -> Node {
    match stmt {
        Statement::Select(s) => select_stmt_to_node(s),
        _ => Node::Text { value: format!("{}", stmt) },
    }
}

fn select_stmt_to_node(s: &SelectStatement<Raw>) -> Node {
    let query = &s.query;
    select_query_to_node(query)
}

fn select_query_to_node(query: &Query<Raw>) -> Node {
    match &query.body {
        SetExpr::Select(select) => select_to_node(select, query),
        _ => Node::Text { value: format!("{}", query) },
    }
}

fn select_to_node(select: &Select<Raw>, query: &Query<Raw>) -> Node {
    let mut clauses = Vec::new();

    // SELECT [DISTINCT] cols
    let cols: Vec<Node> = select.projection.iter().map(select_item_to_node).collect();
    clauses.push(Clause {
        keyword: "SELECT".into(),
        body: Some(Box::new(Node::List { items: cols, separator: None })),
    });

    // FROM
    if !select.from.is_empty() {
        let from_items: Vec<Node> = select.from.iter().map(table_with_joins_to_node).collect();
        clauses.push(Clause {
            keyword: "FROM".into(),
            body: Some(Box::new(Node::List { items: from_items, separator: None })),
        });
    }

    // WHERE
    if let Some(selection) = &select.selection {
        clauses.push(Clause {
            keyword: "WHERE".into(),
            body: Some(Box::new(expr_to_node(selection))),
        });
    }

    // GROUP BY
    if !select.group_by.is_empty() {
        let group_items: Vec<Node> = select.group_by.iter().map(expr_to_node).collect();
        clauses.push(Clause {
            keyword: "GROUP BY".into(),
            body: Some(Box::new(Node::List { items: group_items, separator: None })),
        });
    }

    // HAVING
    if let Some(having) = &select.having {
        clauses.push(Clause {
            keyword: "HAVING".into(),
            body: Some(Box::new(expr_to_node(having))),
        });
    }

    // ORDER BY
    if !query.order_by.is_empty() {
        let order_items: Vec<Node> = query.order_by.iter().map(order_by_expr_to_node).collect();
        clauses.push(Clause {
            keyword: "ORDER BY".into(),
            body: Some(Box::new(Node::List { items: order_items, separator: None })),
        });
    }

    // LIMIT
    if let Some(limit) = &query.limit {
        clauses.push(Clause {
            keyword: "LIMIT".into(),
            body: Some(Box::new(expr_to_node(limit))),
        });
    }

    Node::Clauses { items: clauses }
}

fn select_item_to_node(item: &SelectItem<Raw>) -> Node {
    match item {
        SelectItem::Expr { expr, alias } => {
            let expr_node = expr_to_node(expr);
            match alias {
                None => expr_node,
                Some(alias) => Node::Concat {
                    items: vec![
                        expr_node,
                        Node::Text { value: " ".into() },
                        Node::Keyword { value: "AS".into() },
                        Node::Text { value: " ".into() },
                        ident_to_node(alias),
                    ],
                },
            }
        }
        SelectItem::Wildcard => Node::Text { value: "*".into() },
    }
}

fn table_with_joins_to_node(twj: &TableWithJoins<Raw>) -> Node {
    // For now, just format the relation; joins fall back to text
    table_factor_to_node(&twj.relation)
}

fn table_factor_to_node(factor: &TableFactor<Raw>) -> Node {
    match factor {
        TableFactor::Table { name, alias, .. } => {
            let name_node = Node::Identifier { value: name.to_string(), quote: None };
            match alias {
                None => name_node,
                Some(alias) => Node::Concat {
                    items: vec![
                        name_node,
                        Node::Text { value: " ".into() },
                        Node::Keyword { value: "AS".into() },
                        Node::Text { value: " ".into() },
                        Node::Identifier { value: alias.name.to_string(), quote: None },
                    ],
                },
            }
        }
        TableFactor::Derived { subquery, alias, .. } => {
            let inner = select_query_to_node(subquery);
            let wrapped = Node::Wrap {
                keyword: None,
                open: "(".into(),
                content: Box::new(inner),
                close: ")".into(),
            };
            match alias {
                None => wrapped,
                Some(alias) => Node::Concat {
                    items: vec![
                        wrapped,
                        Node::Text { value: " ".into() },
                        Node::Identifier { value: alias.name.to_string(), quote: None },
                    ],
                },
            }
        }
        _ => Node::Text { value: format!("{factor}") },
    }
}

pub fn expr_to_node(expr: &Expr<Raw>) -> Node {
    match expr {
        Expr::Identifier(names) => {
            Node::Identifier { value: names.iter().map(|i| i.to_string()).collect::<Vec<_>>().join("."), quote: None }
        }
        Expr::Value(v) => Node::Literal { value: v.to_string() },
        Expr::BinaryOp { left, op, right } => Node::Infix {
            op: op.to_string(),
            items: vec![expr_to_node(left), expr_to_node(right)],
        },
        Expr::UnaryOp { op, expr } => Node::Concat {
            items: vec![
                Node::Keyword { value: op.to_string() },
                Node::Text { value: " ".into() },
                expr_to_node(expr),
            ],
        },
        Expr::IsNull(e) => Node::Concat {
            items: vec![
                expr_to_node(e),
                Node::Text { value: " ".into() },
                Node::Keyword { value: "IS NULL".into() },
            ],
        },
        Expr::IsNotNull(e) => Node::Concat {
            items: vec![
                expr_to_node(e),
                Node::Text { value: " ".into() },
                Node::Keyword { value: "IS NOT NULL".into() },
            ],
        },
        Expr::Function(f) => {
            let args: Vec<Node> = f.args.iter().map(|a| match a {
                FunctionArg::Unnamed(expr) => expr_to_node(expr),
                FunctionArg::Named { name, arg } => Node::Concat {
                    items: vec![
                        ident_to_node(name),
                        Node::Text { value: " => ".into() },
                        expr_to_node(arg),
                    ],
                },
            }).collect();
            Node::Wrap {
                keyword: Some(f.name.to_string()),
                open: "(".into(),
                content: Box::new(Node::List { items: args, separator: None }),
                close: ")".into(),
            }
        }
        Expr::Nested(e) => Node::Wrap {
            keyword: None,
            open: "(".into(),
            content: Box::new(expr_to_node(e)),
            close: ")".into(),
        },
        Expr::Subquery(q) => Node::Wrap {
            keyword: None,
            open: "(".into(),
            content: Box::new(select_query_to_node(q)),
            close: ")".into(),
        },
        _ => Node::Text { value: format!("{expr}") },
    }
}

fn order_by_expr_to_node(expr: &OrderByExpr<Raw>) -> Node {
    let mut items = vec![expr_to_node(&expr.expr)];
    if let Some(asc) = expr.asc {
        items.push(Node::Keyword {
            value: if asc { "ASC".into() } else { "DESC".into() },
        });
    }
    if items.len() == 1 {
        items.remove(0)
    } else {
        Node::Concat {
            items: items.into_iter().flat_map(|n| vec![Node::Text { value: " ".into() }, n]).skip(1).collect(),
        }
    }
}

fn ident_to_node(ident: &Ident) -> Node {
    Node::Identifier { value: ident.to_string(), quote: None }
}

The `pg_query` crate returns a protobuf AST. The top-level statement is `pg_query::protobuf::RawStmt`. Navigate via `.stmt` which is `Option<pg_query::protobuf::Node>`, and `.node` within that which is `Option<pg_query::protobuf::node::Node>` (an enum of all statement types).

```rust
// ~/code/sqlfmtng/crates/dialect-postgres/src/convert.rs
use pg_query::protobuf::{
    a_const, node::Node as PgNode, A_Const, ColumnRef, FuncCall, JoinExpr,
    RangeSubselect, RangeVar, RawStmt, ResTarget, SelectStmt, TypeCast,
};
use sqlfmt_ir::{Clause, Node};

pub fn stmt_to_node(raw: &RawStmt) -> Option<Node> {
    let inner = raw.stmt.as_ref()?.node.as_ref()?;
    match inner {
        PgNode::SelectStmt(s) => Some(select_to_node(s)),
        _ => Some(Node::Text { value: format!("{inner:?}") }),
    }
}

fn select_to_node(s: &SelectStmt) -> Node {
    let mut clauses = Vec::new();

    // SELECT cols
    if !s.target_list.is_empty() {
        let cols: Vec<Node> = s.target_list.iter().filter_map(|n| {
            if let Some(PgNode::ResTarget(t)) = n.node.as_ref() {
                Some(res_target_to_node(t))
            } else {
                None
            }
        }).collect();
        clauses.push(Clause {
            keyword: "SELECT".into(),
            body: Some(Box::new(Node::List { items: cols, separator: None })),
        });
    }

    // FROM
    if !s.from_clause.is_empty() {
        let from_items: Vec<Node> = s.from_clause.iter()
            .filter_map(|n| n.node.as_ref().map(from_node_to_node))
            .collect();
        clauses.push(Clause {
            keyword: "FROM".into(),
            body: Some(Box::new(Node::List { items: from_items, separator: None })),
        });
    }

    // WHERE
    if let Some(where_node) = &s.where_clause {
        if let Some(inner) = &where_node.node {
            clauses.push(Clause {
                keyword: "WHERE".into(),
                body: Some(Box::new(pg_node_to_node(inner))),
            });
        }
    }

    // GROUP BY
    if !s.group_clause.is_empty() {
        let group_items: Vec<Node> = s.group_clause.iter()
            .filter_map(|n| n.node.as_ref().map(pg_node_to_node))
            .collect();
        clauses.push(Clause {
            keyword: "GROUP BY".into(),
            body: Some(Box::new(Node::List { items: group_items, separator: None })),
        });
    }

    // HAVING
    if let Some(having) = &s.having_clause {
        if let Some(inner) = &having.node {
            clauses.push(Clause {
                keyword: "HAVING".into(),
                body: Some(Box::new(pg_node_to_node(inner))),
            });
        }
    }

    // ORDER BY
    if !s.sort_clause.is_empty() {
        let order_items: Vec<Node> = s.sort_clause.iter()
            .filter_map(|n| n.node.as_ref().map(pg_node_to_node))
            .collect();
        clauses.push(Clause {
            keyword: "ORDER BY".into(),
            body: Some(Box::new(Node::List { items: order_items, separator: None })),
        });
    }

    // LIMIT
    if let Some(limit) = &s.limit_count {
        if let Some(inner) = &limit.node {
            clauses.push(Clause {
                keyword: "LIMIT".into(),
                body: Some(Box::new(pg_node_to_node(inner))),
            });
        }
    }

    Node::Clauses { items: clauses }
}

fn res_target_to_node(t: &ResTarget) -> Node {
    let val = t.val.as_ref()
        .and_then(|n| n.node.as_ref())
        .map(pg_node_to_node)
        .unwrap_or_else(|| Node::Text { value: "?".into() });

    if t.name.is_empty() {
        val
    } else {
        Node::Concat {
            items: vec![
                val,
                Node::Text { value: " ".into() },
                Node::Keyword { value: "AS".into() },
                Node::Text { value: " ".into() },
                Node::Identifier { value: t.name.clone(), quote: None },
            ],
        }
    }
}

fn from_node_to_node(n: &PgNode) -> Node {
    match n {
        PgNode::RangeVar(rv) => range_var_to_node(rv),
        PgNode::JoinExpr(j) => join_to_node(j),
        PgNode::RangeSubselect(rs) => range_subselect_to_node(rs),
        _ => Node::Text { value: format!("{n:?}") },
    }
}

fn range_var_to_node(rv: &RangeVar) -> Node {
    let name = if rv.schemaname.is_empty() {
        rv.relname.clone()
    } else {
        format!("{}.{}", rv.schemaname, rv.relname)
    };
    let node = Node::Identifier { value: name, quote: None };
    if let Some(alias) = &rv.alias {
        Node::Concat {
            items: vec![
                node,
                Node::Text { value: " ".into() },
                Node::Keyword { value: "AS".into() },
                Node::Text { value: " ".into() },
                Node::Identifier { value: alias.aliasname.clone(), quote: None },
            ],
        }
    } else {
        node
    }
}

fn join_to_node(j: &JoinExpr) -> Node {
    // Fallback to text for joins for now
    Node::Text { value: format!("{j:?}") }
}

fn range_subselect_to_node(rs: &RangeSubselect) -> Node {
    let inner = rs.subquery.as_ref()
        .and_then(|n| n.node.as_ref())
        .map(|n| match n {
            PgNode::SelectStmt(s) => select_to_node(s),
            _ => Node::Text { value: format!("{n:?}") },
        })
        .unwrap_or_else(|| Node::Text { value: "?".into() });

    Node::Wrap {
        keyword: None,
        open: "(".into(),
        content: Box::new(inner),
        close: ")".into(),
    }
}

fn pg_node_to_node(n: &PgNode) -> Node {
    match n {
        PgNode::ColumnRef(cr) => column_ref_to_node(cr),
        PgNode::AConst(ac) => a_const_to_node(ac),
        PgNode::FuncCall(fc) => func_call_to_node(fc),
        PgNode::TypeCast(tc) => type_cast_to_node(tc),
        PgNode::BoolExpr(be) => {
            let op = match be.boolop {
                0 => "AND",
                1 => "OR",
                _ => "NOT",
            };
            let items: Vec<Node> = be.args.iter()
                .filter_map(|n| n.node.as_ref().map(pg_node_to_node))
                .collect();
            Node::Infix { op: op.into(), items }
        }
        PgNode::AExpr(ae) => {
            let op = ae.name.first()
                .and_then(|n| n.node.as_ref())
                .map(|n| match n {
                    PgNode::String(s) => s.sval.clone(),
                    _ => format!("{n:?}"),
                })
                .unwrap_or_default();
            let left = ae.lexpr.as_ref().and_then(|n| n.node.as_ref()).map(pg_node_to_node);
            let right = ae.rexpr.as_ref().and_then(|n| n.node.as_ref()).map(pg_node_to_node);
            match (left, right) {
                (Some(l), Some(r)) => Node::Infix { op, items: vec![l, r] },
                (Some(l), None) => Node::Concat {
                    items: vec![l, Node::Text { value: format!(" {op}") }],
                },
                _ => Node::Text { value: op },
            }
        }
        PgNode::SubLink(sl) => {
            let inner = sl.subselect.as_ref()
                .and_then(|n| n.node.as_ref())
                .map(|n| match n {
                    PgNode::SelectStmt(s) => select_to_node(s),
                    _ => Node::Text { value: format!("{n:?}") },
                })
                .unwrap_or_else(|| Node::Text { value: "?".into() });
            Node::Wrap { keyword: None, open: "(".into(), content: Box::new(inner), close: ")".into() }
        }
        PgNode::SelectStmt(s) => select_to_node(s),
        _ => Node::Text { value: format!("{n:?}") },
    }
}

fn column_ref_to_node(cr: &ColumnRef) -> Node {
    let parts: Vec<String> = cr.fields.iter().filter_map(|n| {
        n.node.as_ref().map(|n| match n {
            PgNode::String(s) => s.sval.clone(),
            PgNode::AStar(_) => "*".into(),
            _ => format!("{n:?}"),
        })
    }).collect();
    Node::Identifier { value: parts.join("."), quote: None }
}

fn a_const_to_node(ac: &A_Const) -> Node {
    let val = match &ac.val {
        Some(a_const::Val::Ival(i)) => i.ival.to_string(),
        Some(a_const::Val::Fval(f)) => f.fval.clone(),
        Some(a_const::Val::Sval(s)) => format!("'{}'", s.sval.replace('\'', "''")),
        Some(a_const::Val::Bval(b)) => if b.bval { "TRUE".into() } else { "FALSE".into() },
        None => "NULL".into(),
        _ => format!("{ac:?}"),
    };
    Node::Literal { value: val }
}

fn func_call_to_node(fc: &FuncCall) -> Node {
    let name = fc.funcname.iter().filter_map(|n| {
        n.node.as_ref().map(|n| match n {
            PgNode::String(s) => s.sval.clone(),
            _ => format!("{n:?}"),
        })
    }).collect::<Vec<_>>().join(".");

    let args: Vec<Node> = fc.args.iter()
        .filter_map(|n| n.node.as_ref().map(pg_node_to_node))
        .collect();

    if args.is_empty() && fc.agg_star {
        return Node::Wrap {
            keyword: Some(name),
            open: "(".into(),
            content: Box::new(Node::Text { value: "*".into() }),
            close: ")".into(),
        };
    }

    Node::Wrap {
        keyword: Some(name),
        open: "(".into(),
        content: Box::new(Node::List { items: args, separator: None }),
        close: ")".into(),
    }
}

fn type_cast_to_node(tc: &TypeCast) -> Node {
    let arg = tc.arg.as_ref()
        .and_then(|n| n.node.as_ref())
        .map(pg_node_to_node)
        .unwrap_or_else(|| Node::Text { value: "?".into() });

    let type_name = tc.type_name.as_ref()
        .map(|t| t.names.iter().filter_map(|n| {
            n.node.as_ref().map(|n| match n {
                PgNode::String(s) => s.sval.clone(),
                _ => format!("{n:?}"),
            })
        }).collect::<Vec<_>>().join("."))
        .unwrap_or_else(|| "?".into());

    Node::Wrap {
        keyword: Some("CAST".into()),
        open: "(".into(),
        content: Box::new(Node::Concat {
            items: vec![
                arg,
                Node::Text { value: " ".into() },
                Node::Keyword { value: "AS".into() },
                Node::Text { value: " ".into() },
                Node::Identifier { value: type_name, quote: None },
            ],
        }),
        close: ")".into(),
    }
}

// ~/code/sqlfmtng/src/main.rs
use clap::{Parser, ValueEnum};
use sqlfmt_core::format_sql;
use sqlfmt_render::{CaseMode, RenderOpts};
use std::io::{self, Read};

#[derive(Debug, Clone, Copy, ValueEnum)]
enum DialectArg {
    Materialize,
    Postgres,
}

#[derive(Debug, Clone, Copy, ValueEnum)]
enum CaseArg {
    Upper,
    Lower,
    Title,
}

#[derive(Parser, Debug)]
#[command(name = "sqlfmt", about = "SQL formatter")]
struct Args {
    /// SQL dialect
    #[arg(long, default_value = "materialize")]
    dialect: DialectArg,

    /// Target line width
    #[arg(long, default_value_t = 60)]
    width: usize,

    /// Spaces per indent level
    #[arg(long, default_value_t = 4)]
    tab_width: usize,

    /// Use tab characters instead of spaces
    #[arg(long)]
    use_tabs: bool,

    /// Keyword case
    #[arg(long, default_value = "upper")]
    case: CaseArg,

    /// SQL statement (reads from stdin if omitted)
    #[arg(long)]
    stmt: Option<String>,
}

fn main() {
    let args = Args::parse();

    let sql = match &args.stmt {
        Some(s) => s.clone(),
        None => {
            let mut buf = String::new();
            io::stdin().read_to_string(&mut buf).expect("failed to read stdin");
            buf
        }
    };

    let opts = RenderOpts {
        line_width: args.width,
        use_tabs: args.use_tabs,
        tab_width: args.tab_width,
        case: match args.case {
            CaseArg::Upper => CaseMode::Upper,
            CaseArg::Lower => CaseMode::Lower,
            CaseArg::Title => CaseMode::Title,
        },
    };

    let result = match args.dialect {
        DialectArg::Materialize => {
            format_sql(&dialect_materialize::MaterializeDialect, &sql, &opts)
        }
        DialectArg::Postgres => {
            format_sql(&dialect_postgres::PostgresDialect, &sql, &opts)
        }
    };

    match result {
        Ok(formatted) => println!("{formatted}"),
        Err(e) => {
            eprintln!("error: {e}");
            std::process::exit(1);
        }
    }
}
```

// ~/code/sqlfmtng/web/src/lib.rs
use sqlfmt_core::format_sql;
use sqlfmt_render::{CaseMode, RenderOpts};
use wasm_bindgen::prelude::*;

/// Format SQL and return the result.
/// dialect: "materialize" | "postgres"
/// case: "upper" | "lower" | "title"
#[wasm_bindgen]
pub fn fmt(
    sql: &str,
    dialect: &str,
    width: usize,
    tab_width: usize,
    use_tabs: bool,
    case: &str,
) -> Result<String, JsValue> {
    let opts = RenderOpts {
        line_width: width,
        use_tabs,
        tab_width,
        case: match case {
            "lower" => CaseMode::Lower,
            "title" => CaseMode::Title,
            _ => CaseMode::Upper,
        },
    };

    let result = match dialect {
        "postgres" => format_sql(&dialect_postgres::PostgresDialect, sql, &opts),
        _ => format_sql(&dialect_materialize::MaterializeDialect, sql, &opts),
    };

    result.map_err(|e| JsValue::from_str(&e.to_string()))
}
